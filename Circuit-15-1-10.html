\(\Large \displaystyle F(s) = \frac{V_0}{L} \frac{1}{s^2 + 2 \ \alpha \ s + \omega_0^2} \)

\(\Large \displaystyle = \frac{V_0}{L} \frac{1}{s^2 + 2 \ \alpha \ s + \alpha^2} \)

\(\Large \displaystyle F(s) = \frac{V_0}{L} \frac{1}{( s+ \alpha)^2} \)
